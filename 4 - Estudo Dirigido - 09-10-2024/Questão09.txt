// 9. Contar caracteres em uma string
function contarCaracteres(str) {
    let contagem = {};
    for (let char of str) {
        contagem[char] = (contagem[char] || 0) + 1; // Incrementa a contagem de cada caractere
    }
    return contagem;
}
